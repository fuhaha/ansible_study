# Commands used in Peter's 2016-08-29 demo
ansible all -m ping
ansible-playbook -K apt-loop.yml
# Commands from VM to verify that pkg's were installed
dpkg -l {git,nginx}
rpm -q {git,nginx}
# Run the same command using Ansible's 'command' module
# -a --args (arguments to pass to the module)
ansible ubuntu_vm -m command -a 'dpkg -l git'
ansible ubuntu_vm -m command -a 'dpkg -l nginx'
ansible rhel_vm -m command -a 'rpm -q git'
ansible rhel_vm -m command -a 'rpm -q nginx'

# Since 'command' is the default module, you don't need to specify it
ansible ubuntu_vm -a 'dpkg -l git'
ansible ubuntu_vm -a 'dpkg -l nginx'
ansible rhel_vm -a 'rpm -q git'
ansible rhel_vm -a 'rpm -q nginx'
